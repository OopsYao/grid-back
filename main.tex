\documentclass{article}
\usepackage{tikz}

\usetikzlibrary{calc}

\newcommand{\gridsize}{14bp}
% Offset that matches the grid
\newcommand{\southwestoffsetx}{36.7bp}
\newcommand{\southwestoffsety}{41.2bp}
\newcommand{\gridgroupsize}{3}


\begin{document}
\thispagestyle{empty}
\begin{tikzpicture}[remember picture,overlay,color=black!20]%
  \draw[help lines,color=black!20] (current page.south west) grid[step=\gridsize] (current page.north east);
  \foreach \y in {0, ..., 20}
    {%
      % Left border
      \filldraw ([xshift=-1pt,yshift=-1pt]$(current page.south west) + (\southwestoffsetx, \southwestoffsety)
        + (0, \gridsize *\gridgroupsize * \y)$)
      rectangle ++(2pt, 2pt);
      % Sidenote line
      \filldraw ($(current page.south west) + (\southwestoffsetx, \southwestoffsety)
      + (1 * \gridsize + 9 * \gridsize * \gridgroupsize, \gridsize * \gridgroupsize * \y)$)
      circle (1pt);
      % Right border
      \filldraw ([xshift=-1pt,yshift=-1pt]$(current page.south west) + (\southwestoffsetx, \southwestoffsety)
        + (2 * \gridsize + 12 * \gridsize * \gridgroupsize, \gridsize * \gridgroupsize * \y)$)
      rectangle ++(2pt, 2pt);
    }
\end{tikzpicture}
\end{document}
