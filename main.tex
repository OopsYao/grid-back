\documentclass{article}
\usepackage{tikz}

\usetikzlibrary{calc}

\newcommand{\gridsize}{14bp}
% Offset that matches the grid
\newcommand{\southwestoffsetx}{36.7bp}
\newcommand{\southwestoffsety}{41.2bp}
\newcommand{\gridgroupsize}{3}


\begin{document}
\thispagestyle{empty}
\begin{tikzpicture}[remember picture,overlay,color=black!20]%
  \draw[help lines,color=black!20] (current page.south west) grid[step=\gridsize] (current page.north east);
  \foreach \y in {0, ..., 20}
    {%
      % Section
      \filldraw ($(current page.south west) + (\southwestoffsetx, \southwestoffsety)
      + (0, \gridsize *\gridgroupsize * \y)$) circle (0.7pt);
      % Subsection
      \filldraw ($(current page.south west) + (\southwestoffsetx, \southwestoffsety)
      + (1 * \gridsize * \gridgroupsize, \gridsize * \gridgroupsize * \y)$) circle (0.5pt);
      % Subsubsection
      \filldraw ($(current page.south west) + (\southwestoffsetx, \southwestoffsety)
      + (2 * \gridsize * \gridgroupsize, \gridsize * \gridgroupsize * \y)$) circle (0.3pt);
      % Midline
      \filldraw ($(current page.south) + (0, \southwestoffsety)
      + (10.7bp, \gridsize * \gridgroupsize * \y)$) circle (0.7pt);
    }
\end{tikzpicture}
\end{document}
